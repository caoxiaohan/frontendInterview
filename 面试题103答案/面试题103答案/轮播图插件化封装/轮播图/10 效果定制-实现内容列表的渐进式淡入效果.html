var oSlide = new Slider("#slideCustom"),
    done = function(){ oSlide.done(); };           
    
oSlide.init(function(cur, old, action){

    var $lists = this.lists,
        max = $lists.length - 1;
        cfg = this.cfg,
        must = action === "must",
        duration = must ? 0 : cfg.duration,
        start = cfg.scrollLen * cur,
        end = cfg.scrollLen * (cur + 1) - 1,
        delay = 0,
        step = must ? 0 : 40;

    end = end > max ? max : end;
    
    $lists.hide();
    for(var i = start; i <= end; i++){

        $lists.eq(i).delay(delay).fadeIn(duration, cfg.easing, i === end ? done : undefined);
        delay += step;
    }
});
